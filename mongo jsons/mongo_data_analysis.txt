            db['Species'].aggregate([
            {
                $lookup:{
                    from: "Batch",

                }

                {$match:{"species":{$size:0}}}
            
            ])

    --3. What is the current seed inventory, sorted alphabetically by common name?

-- Education module
    --9. What are the 3 most popular eventsand what was the attendence of each?

        db['Events'].aggregate([
        {
            $group:{
            _id:"$event_name",
                attendence:{
                    $sum:{
                $add:[{$size: "$guest_attendance"},{$size: "$member_attendance"} ]
                }   
            }
            } 
        },
        {$sort:{attendence:-1}},
        {$limit:3}

        ])

    --10. Return a list of every guest and their emails who has attended at least 1 event

    db['Events'].aggregate([
  {$unwind: "$guest_attendance"},
  {$project: {_id:0, name: "$guest_attendance.name", email: "$guest_attendance.email"}}
])
